2#Get least expensive product name and price 
SELECT products.product_name , products_price.product_price  
FROM products, products_price
WHERE products.product_id = products_price.product_id
  AND products_price.product_price in (SELECT MIN(product_price) FROM  products_price)

3#Count number of products with price less than 5
